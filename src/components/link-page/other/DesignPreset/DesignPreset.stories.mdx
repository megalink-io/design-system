# DesignPreset

A clickable element that serves as a preview of the link page's design presets.

export const getProps = (id) => ({
  selected: boolean('Selected', false, id),
  onClick: action('onClick', id),
  design: {
    document: {
      backgroundVariant: select('Design > Document > Background variant', ['color', 'line_gradient'], 'color', id),
      backgroundColor: text('Design > Document > Background color', '#fff', id),
      backgroundLineGradient: text('Design > Document > Background line gradient', '', id),
    },
    elements: {
      default: {
        alignment: select('Design > Elements > Default > Alignment', ['left', 'center', 'right'], 'center', id),
        fontFamily: select('Design > Elements > Default > Font family', ['poppins', 'open_sans', 'gelasio', 'roboto_mono'], 'poppins', id),
        fontColor: text('Design > Elements > Default > Font color', '#fff', id),
        shape: select('Design > Elements > Default > Shape', ['angular', 'rounded', 'round'], 'angular', id),
        backgroundVariant: select('Design > Elements > Default > Background variant', ['filled', 'outlined'], 'filled', id),
        backgroundColor: text('Design > Elements > Default > Background color', '#313131', id),
        shadowVariant: select('Design > Elements > Default > Shadow variant', ['none', 'soft', 'hard'], 'none', id),
        shadowColor: text('Design > Elements > Default > Shadow color', '', id),
      }
    }
  }
});
