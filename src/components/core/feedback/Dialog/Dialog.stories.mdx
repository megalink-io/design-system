# Dialog

<Description of={Dialog} />

> Click on one of the buttons to open a sample dialog.

export const Template = args => {
  const [open, setOpen] = useState(false);
  const getButtonColor = () => {
    switch (args.variant) {
      case 'confirmation':
        return 'blue';
      case 'information':
        return 'gray';
      default:
        return 'red';
    }
  };
  const getButtonLabel = () => {
    switch (args.variant) {
      case 'confirmation':
        return 'Enable';
      case 'information':
        return 'Continue';
      case 'destructive':
        return 'Delete';
      default:
        return 'OK';
    }
  };
  return (
    <>
      <ActionButton
        label={`${args.variant.charAt(0).toUpperCase()}${args.variant.slice(1)}`}
        variant="tertiary"
        color={getButtonColor()}
        type="button"
        onClick={() => {
          setOpen(true);
        }}
      />
      <Dialog
        open={open}
        actionButton={{
          label: getButtonLabel(),
          type: 'button',
          onClick: () => {
            setOpen(false);
          },
        }}
        cancelButton={
          args.variant !== 'error'
            ? {
                label: 'Cancel',
                type: 'button',
                onClick: () => {
                  setOpen(false);
                },
              }
            : undefined
        }
        {...args}
      />
    </>
  );
};

export const getProps = id => ({
  variant: id,
  heading: 'This is the heading',
  message: `This is just a great example text for the ${id} dialog message.`,
});
