# FloatingButton

A floating button with an icon at the bottom right of the screen used to navigate or perform individual actions.

<Preview className="fullframe" withToolbar>
  <Story name="Share button">
    <FloatingButton
      icon={<ShareSVG />}

      type="button"
      color={select('Color', ['gray', 'yellow', 'red', 'blue', 'green'], 'blue', 'share_button')}
      onClick={action('onClick', 'share_button')}
      loading={boolean('Loading', false, 'share_button')}
      disabled={boolean('Disabled', false, 'share_button')}
    />
  </Story>
</Preview>

## Code example

```tsx
import { FloatingButton } from '@mds/components/core/actions';
```

```tsx
<FloatingButton icon={<DeleteSVG />} label="Delete" color="blue" type="button" onClick={() => handleClick()} />
<FloatingButton icon={<GearSVG />} label="Settings" color="blue" type="link" to="/settings" />
```
