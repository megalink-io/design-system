# IconButton

A button with an icon that is used to navigate or perform small actions.

export const getProps = id => ({
  icon: <DeleteSVG />,
  label: 'Delete',
  variant: id,
  type: 'button',
  color: select('Color', ['white', 'black', 'gray', 'yellow', 'red', 'blue', 'green'], 'blue', id),
  onClick: action('onClick', id),
  loading: boolean('Loading', false, id),
  disabled: boolean('Disabled', false, id),
});

## Code example

```tsx
import { IconButton } from '@mds/components/basic/actions';
```

```tsx
<IconButton icon={<DeleteSVG />} label="Delete" variant="tertiary" type="button" onClick={() => handleClick()} />
<IconButton icon={<GearSVG />} label="Settings" variant="tertiary" type="link" to="/settings" />
```
